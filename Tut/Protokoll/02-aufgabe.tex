#include "fir.h"

void fir (
  data_t result[20],
  data_t n
  ) {

  int i;
  *result = 0;
  *(result+1) = 1;
  Shift_Accum_Loop: for (i=2;i<20;i++) {
	  *(result+i) =  *(result+i-1) +  *(result+i-2);
  }
}
